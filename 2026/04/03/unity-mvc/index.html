unity中的MVC

首先，什么是MVC，MVC由Model，View，Controller组成。

view从Model层中获取数据，不改动Model层，接收用户的输入传给Controller层。

Controller层接收并解析用户输入，更新和使用Model的数据，管理Unity的场景，不存储核心数据

Model层存储游戏的数据结构与逻辑并将更新后的数据发给view

首先设计一个事件中心

using System.Collections.Generic;
using UnityEngine;
using UnityEngine.Events;

// 事件响应空接口，用于支持可有可无的参数类型
public interface IEventInfo
{
}

// 带参数事件响应
public class EventInfo<T> : IEventInfo
{
    public UnityAction<T> actions;

    public EventInfo(UnityAction<T> action)
    {
        actions += action;
    }
}

// 无参数事件响应
public class EventInfo : IEventInfo
{
    public UnityAction actions;
    public EventInfo(UnityAction action)
    {
        actions += action;
    }
}
public class EventCenter
{
    // 数据结构
    private Dictionary<string, IEventInfo> _eventDic;

    public void init()
    {
        _eventDic = new Dictionary<string, IEventInfo>();
    }

    // 添加带参数事件的监听
    public void AddEventListener<T>(string name, UnityAction<T> action)
    {
        // 旧事件
        if (_eventDic.ContainsKey(name))
            (_eventDic[name] as EventInfo<T>).actions += action;
        // 新事件
        else
        {
            Debug.Log("已注册"+name);
            _eventDic.Add(name, new EventInfo<T>(action));
            Debug.Log("@@@"+_eventDic.Count);
        }
    }

    // 添加无参数事件的监听
    public void AddEventListener(string name, UnityAction action)
    {
        if (_eventDic.ContainsKey(name))
            (_eventDic[name] as EventInfo).actions += action;
        else
            _eventDic.Add(name, new EventInfo(action));
    }

    // 移除带参数事件的监听
    public void RemoveEventListener<T>(string name, UnityAction<T> action)
    {
        if (_eventDic.ContainsKey(name))
            (_eventDic[name] as EventInfo<T>).actions -= action;
    }

    // 移除无参数事件的监听
    public void RemoveEventListener(string name, UnityAction action)
    {
        if (_eventDic.ContainsKey(name))
            (_eventDic[name] as EventInfo).actions -= action;
    }

    // 分发带参数的事件
    public void EventTrigger<T>(string name, T info)
    {

        if (_eventDic.ContainsKey(name))
        {
            Debug.Log((_eventDic[name] as EventInfo<T>).actions != null);
            if ((_eventDic[name] as EventInfo<T>).actions != null)
            {Debug.Log("一份发");
                (_eventDic[name] as EventInfo<T>).actions.Invoke(info);
                
            }
        }
            
    }

    // 分发无参数的事件
    public void EventTrigger(string name)
    {
        if (_eventDic.ContainsKey(name))
            if ((_eventDic[name] as EventInfo).actions != null)
                (_eventDic[name] as EventInfo).actions.Invoke();
    }

    // 清空事件监听
    // 主要用于场景切换时防止内存泄漏
    public void Clear()
    {
        _eventDic.Clear();
    }
}

在FGUI里做好UI并导出

using System.Collections;
using System.Collections.Generic;
using FairyGUI;
using UnityEditor.SceneManagement;
using UnityEngine;

public class GameMain : Window
{
    private GObject[] children;
    private ListItemRenderer itemRenderer;
    int i = 0;
    private Dictionary<int, GComponent> dic;
    protected override void OnInit()
    {
        base.OnInit();
        dic = new Dictionary<int, GComponent>();
        children = GetChildren();
        StartManager.I.eve.AddEventListener<date>(Name.SetPosition, SetPosition);
        StartManager.I.eve.AddEventListener<date>(Name.AddPlane, AddPlane);
        GList list = GetChild("n4").asList;
        list.defaultItem = "ui://Package1/bg";
        list.SetVirtualAndLoop();  
        list.itemRenderer = (index, obj) => { };
        list.numItems = 2;  
        ScrollPane sp = list.scrollPane;  
        float speed = 100f;
        // 用 GTween 驱动滚动位置，从 0 滚到最大，线性匀速，无限循环  
        GTween.To(0f, sp.contentHeight, 500f)  
            .SetEase(EaseType.Linear)  
            .SetRepeat(-1, false)  // -1 = 无限循环  
            .OnUpdate(() =>  
            {  
                sp.posY += speed * Time.deltaTime;
            })  
            .SetTarget(list);
        
        GameManager.I.init();
    }

    public void AddPlane(date y)
    {
        GComponent p= UIPackage.CreateObject("Package1", "plane") as GComponent;
        p.x = y.p;
        p.y = y.y;
        AddChild(p);
        dic.Add(y.id,p);
        
    }
    public void SetPosition(date newPos)
    {
        
            Debug.Log("d.id"+newPos.id+"d.p"+newPos.p);
            if (dic.ContainsKey(newPos.id))
            {
                dic[newPos.id].x = newPos.p;
                dic[newPos.id].y = newPos.y;
                Debug.Log("id"+newPos.id+"ch"+dic[newPos.id].y);
                
            }
            else
            {
                Debug.Log("lost");
            }
        
    }
    
    protected override void OnUpdate()
    {
        base.OnUpdate();
    }
}

VIEW层代码，提供创建敌人和控制敌人的函数给Model层调用。

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class GameLogic 
{
    private Dictionary<int, date> dic;
    private int id = 1;

    public void Creat()
    {
        dic = new Dictionary<int, date>();
    }
    public void AddPlane(date d)
    {
        dic.Add(d.id,d);
        StartManager.I.eve.EventTrigger(Name.AddPlane,d); 
    }

    public void SetPosition(Queue<date> newPos)
    {
        Debug.Log("new"+newPos.Count);
        while (newPos.Count != 0)
        {
            date d = newPos.Dequeue();
            Debug.Log("d.id"+d.id+"d.p"+d.p);
            if (dic.ContainsKey(d.id))
            {
                dic[d.id].p = d.p;
                dic[d.id].y = d.y;
                Debug.Log("id"+d.id+"ch"+dic[d.id].y);
                StartManager.I.eve.EventTrigger(Name.SetPosition, d);
            }
            else
            {
                Debug.Log("lost");
            }
        }
    }
}

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEditor.SceneManagement;
public class GameManager 
{
    private static GameManager instance;
    private GameLogic game;
    
    public static GameManager I
    {
        get
        {
            if(instance == null)
            {
                instance = new GameManager();
            }
            return instance;
        }
    }
    
    public void init()
    {
        Debug.Log("qqqqq");
        Queue<date> q = new Queue<date>();
        
        game = new GameLogic();
        game.Creat();
        date d = new date(1,1,1);
        game.AddPlane(d);
        
        d = new date(2,1,1);
        game.AddPlane(d);
        
        d = new date(3,1,1);
        game.AddPlane(d);
        
        
        q.Enqueue(new date(1,100,100));
        q.Enqueue(new date(2,200,200));
        q.Enqueue(new date(3,300,300));
        game.SetPosition(q);
    }
}
